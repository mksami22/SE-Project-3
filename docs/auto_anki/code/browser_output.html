def content_formatter(lines):

    

  

  

  

    

      

        #
      

      
Formats the result in html form

    

    

      
    content = ""
    for line in lines:
        parts = line.split(':', 1)  # Breaks the answer into diff parts for hyperlink
        if parts[0] == 'Related Link':
            content += parts[0] + ": <a href=" + parts[1] + ">"+parts[1]+"</a><br><hr>"
        else:
            content += line + "<br>"
    return content
